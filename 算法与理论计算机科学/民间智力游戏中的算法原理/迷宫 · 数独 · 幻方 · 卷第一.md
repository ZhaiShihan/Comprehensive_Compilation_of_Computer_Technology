### 壹  基础迷宫

###### · 说明：
1. 用 “1” 表示墙，用 “0” 表示可走的路，用 “2” 表示走过的路（标记）
2. 输入格式应当如下：
~~~
4（表示行数）
5（表示列数）
01000
00010
11100
00001（输入的地图）
2（起点横坐标）
1（起点纵坐标）
4（终点横坐标）
1（终点纵坐标）

期待的输出：
Go east!
Go east!
Go north!
Go east!
Go east!
Go south!
Go south!
Go west!
Go south!
Go west!
Go west!
Go west!
~~~

###### · 代码实现：
```C++
#include<stdio.h>
#include<deque>
using namespace std;

typedef struct
{
	int x;
	int y;
} Point;
Point current_place, end_place;    //表示当前坐标和终点坐标

deque <char> DQ;

int next(char arr[][100], Point current, Point end)
{
	if(current.x==end.x&&current.y==end.y) return 1;
	else
	{
		arr[current.x][current.y]='2';
		if(arr[current.x+1][current.y]=='0')    //南边可走
		{
			Point t;
			t.x=current.x+1;
			t.y=current.y;
			DQ.push_back('s');
			if(next(arr,t,end)) return 1;
			DQ.pop_back();
		}
		if(arr[current.x-1][current.y]=='0')    //北边可走
		{
			Point t;
			t.x=current.x-1;
			t.y=current.y;
			DQ.push_back('n');
			if(next(arr,t,end)) return 1;
			DQ.pop_back();
		}
		if(arr[current.x][current.y+1]=='0')    //东边可走
			{
			Point t;
			t.x=current.x;
			t.y=current.y+1;
			DQ.push_back('e');
			if(next(arr,t,end)) return 1;
			DQ.pop_back();
		}
		if(arr[current.x][current.y-1]=='0')    //西边可走
		{
			Point t;
			t.x=current.x;
			t.y=current.y-1;
			DQ.push_back('w');
			if(next(arr,t,end)) return 1;
			DQ.pop_back();
		}
		arr[current.x][current.y]='0';
	}
	
	return 0;
}

int main()
{
	char arr[100][100];
	for(int i=0; i<100; i++) for(int j=0; j<100; j++) arr[i][j]='1';
	int m, n;    //行和列
	
	scanf("%d",&m);
	scanf("%d",&n);
	char ch=getchar();
	
	for(int i=1; i<=m; i++)
	{
		for(int j=1; j<=n; j++)
		{
			scanf("%c",&arr[i][j]);
		}
		ch=getchar();
	}
	
	scanf("%d",&current_place.x);
	scanf("%d",&current_place.y);
	scanf("%d",&end_place.x);
	scanf("%d",&end_place.y);
	
	int judge=next(arr, current_place, end_place);
	if(!judge) printf("Maze without solution!\n");

	if(DQ.empty()&&judge==1) printf("Beginning is the end!\n");
	
	while(!DQ.empty())
	{
		char c=DQ.front();
		if(c=='n') printf("Go north!\n");
		else if(c=='s') printf("Go south!\n");
		else if(c=='e') printf("Go east!\n");
		else if(c=='w') printf("Go west!\n");
		else printf("Error!\n");
		DQ.pop_front();
	}

	return 0;	
}
```
